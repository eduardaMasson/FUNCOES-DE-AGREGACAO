SELECT UNIDADE, SUM(m.ESTOQUE * m.PRECO) AS total_estoque
FROM MEDICAMENTOS m

SELECT ID_RECEITA , SUM(total_item)
FROM ITENSRECEITA i 
GROUP BY i.ID_RECEITA 
HAVING SUM(total_item) > 100
